-- 1. ENUMS
-- ============================================================

CREATE TYPE public.app_role AS ENUM ('admin', 'customer');
CREATE TYPE public.expat_type AS ENUM ('new', 'existing');
CREATE TYPE public.feedback_status AS ENUM ('pending', 'submitted');
CREATE TYPE public.quarter_outcome AS ENUM ('on_target', 'below_target', 'penalty', 'incentive');
CREATE TYPE public.question_section AS ENUM ('service_initiation', 'service_delivery', 'driver_quality', 'overall');

-- 2. USERS & ROLES (leveraging Supabase auth.users)
-- ============================================================

-- Profiles table linked to auth.users
CREATE TABLE public.profiles (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

-- User roles table (never store roles on profiles)
CREATE TABLE public.user_roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    role app_role NOT NULL,
    UNIQUE (user_id, role)
);

ALTER TABLE public.user_roles ENABLE ROW LEVEL SECURITY;

-- Security definer function for role checks (avoids RLS recursion)
CREATE OR REPLACE FUNCTION public.has_role(_user_id UUID, _role app_role)
RETURNS BOOLEAN
LANGUAGE sql
STABLE
SECURITY DEFINER
SET search_path = public
AS $$
    SELECT EXISTS (
        SELECT 1
        FROM public.user_roles
        WHERE user_id = _user_id
          AND role = _role
    )
$$;

-- 3. QUARTERS
-- ============================================================

CREATE TABLE public.quarters (
    id VARCHAR(20) PRIMARY KEY,            -- e.g. 'q1-2026'
    label VARCHAR(20) NOT NULL,            -- e.g. 'Q1 2026'
    year INTEGER NOT NULL,
    quarter_number INTEGER NOT NULL CHECK (quarter_number BETWEEN 1 AND 4),
    is_active BOOLEAN NOT NULL DEFAULT false,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE public.quarters ENABLE ROW LEVEL SECURITY;

-- Ensure only one active quarter at a time
CREATE UNIQUE INDEX idx_quarters_single_active ON public.quarters (is_active) WHERE is_active = true;

-- 4. CUSTOMERS (expatriates using the car rental service)
-- ============================================================

CREATE TABLE public.customers (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    phone VARCHAR(50),
    employee_id VARCHAR(20) NOT NULL UNIQUE,  -- e.g. 'EMP-1042'
    is_active BOOLEAN NOT NULL DEFAULT true,
    created_at DATE NOT NULL DEFAULT CURRENT_DATE,
    allocated_car VARCHAR(100),               -- e.g. 'Toyota Camry'
    start_date DATE,                          -- rental start date
    end_date DATE,                            -- rental end date
    -- Optional link to auth user for customers who log in
    user_id UUID REFERENCES auth.users(id) ON DELETE SET NULL
);

ALTER TABLE public.customers ENABLE ROW LEVEL SECURITY;

CREATE INDEX idx_customers_email ON public.customers (email);
CREATE INDEX idx_customers_user_id ON public.customers (user_id);

-- 5. CUSTOMER QUARTER PROFILES
-- Tracks each customer's expat type per quarter
-- ============================================================

CREATE TABLE public.customer_quarter_profiles (
    id SERIAL PRIMARY KEY,
    customer_id INTEGER NOT NULL REFERENCES public.customers(id) ON DELETE CASCADE,
    quarter_id VARCHAR(20) NOT NULL REFERENCES public.quarters(id) ON DELETE CASCADE,
    expat_type expat_type NOT NULL DEFAULT 'existing',
    UNIQUE (customer_id, quarter_id)
);

ALTER TABLE public.customer_quarter_profiles ENABLE ROW LEVEL SECURITY;

CREATE INDEX idx_cqp_customer ON public.customer_quarter_profiles (customer_id);
CREATE INDEX idx_cqp_quarter ON public.customer_quarter_profiles (quarter_id);

-- 6. SURVEY QUESTIONS
-- ============================================================

CREATE TABLE public.questions (
    id SERIAL PRIMARY KEY,
    question_number INTEGER NOT NULL UNIQUE,
    text TEXT NOT NULL,
    section question_section NOT NULL,
    is_new_expat_only BOOLEAN NOT NULL DEFAULT false,
    is_active BOOLEAN NOT NULL DEFAULT true,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE public.questions ENABLE ROW LEVEL SECURITY;

-- 7. FEEDBACK ASSIGNMENTS
-- Links a customer to a quarter for feedback collection
-- ============================================================

CREATE TABLE public.feedback_assignments (
    id SERIAL PRIMARY KEY,
    quarter_id VARCHAR(20) NOT NULL REFERENCES public.quarters(id) ON DELETE CASCADE,
    customer_id INTEGER NOT NULL REFERENCES public.customers(id) ON DELETE CASCADE,
    status feedback_status NOT NULL DEFAULT 'pending',
    submitted_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    UNIQUE (quarter_id, customer_id)
);

ALTER TABLE public.feedback_assignments ENABLE ROW LEVEL SECURITY;

CREATE INDEX idx_fa_quarter ON public.feedback_assignments (quarter_id);
CREATE INDEX idx_fa_customer ON public.feedback_assignments (customer_id);
CREATE INDEX idx_fa_status ON public.feedback_assignments (status);

-- 8. FEEDBACK RESPONSES
-- Individual question scores per assignment
-- ============================================================

CREATE TABLE public.feedback_responses (
    id SERIAL PRIMARY KEY,
    assignment_id INTEGER NOT NULL REFERENCES public.feedback_assignments(id) ON DELETE CASCADE,
    question_id INTEGER NOT NULL REFERENCES public.questions(id) ON DELETE CASCADE,
    score INTEGER NOT NULL CHECK (score BETWEEN 1 AND 4),
    UNIQUE (assignment_id, question_id)
);

ALTER TABLE public.feedback_responses ENABLE ROW LEVEL SECURITY;

CREATE INDEX idx_fr_assignment ON public.feedback_responses (assignment_id);

-- 9. FEEDBACK COMMENTS
-- Optional free-text comment per assignment
-- ============================================================

CREATE TABLE public.feedback_comments (
    id SERIAL PRIMARY KEY,
    assignment_id INTEGER NOT NULL REFERENCES public.feedback_assignments(id) ON DELETE CASCADE UNIQUE,
    comment TEXT NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE public.feedback_comments ENABLE ROW LEVEL SECURITY;

-- 10. QUARTER REPORTS (aggregated KPI data per quarter)
-- ============================================================

CREATE TABLE public.quarter_reports (
    id SERIAL PRIMARY KEY,
    quarter_id VARCHAR(20) NOT NULL REFERENCES public.quarters(id) ON DELETE CASCADE UNIQUE,
    total_respondents INTEGER NOT NULL DEFAULT 0,
    total_assigned INTEGER NOT NULL DEFAULT 0,
    new_expat_count INTEGER NOT NULL DEFAULT 0,
    -- Service Initiation KPI
    si_avg NUMERIC(4,2) NOT NULL DEFAULT 0,
    si_pct NUMERIC(5,2) NOT NULL DEFAULT 0,
    -- Service Delivery KPI
    sd_avg NUMERIC(4,2) NOT NULL DEFAULT 0,
    sd_pct NUMERIC(5,2) NOT NULL DEFAULT 0,
    -- Driver Quality KPI
    dq_avg NUMERIC(4,2) NOT NULL DEFAULT 0,
    dq_pct NUMERIC(5,2) NOT NULL DEFAULT 0,
    -- Overall Service KPI
    os_avg NUMERIC(4,2) NOT NULL DEFAULT 0,
    os_pct NUMERIC(5,2) NOT NULL DEFAULT 0,
    -- Overall percentage and outcome
    overall_pct NUMERIC(5,2) NOT NULL DEFAULT 0,
    outcome quarter_outcome NOT NULL DEFAULT 'on_target',
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE public.quarter_reports ENABLE ROW LEVEL SECURITY;

-- ROW LEVEL SECURITY POLICIES
-- ============================================================

-- Profiles: users can read their own, admins can read all
CREATE POLICY "Users can view own profile"
    ON public.profiles FOR SELECT
    TO authenticated
    USING (id = auth.uid());

CREATE POLICY "Admins can view all profiles"
    ON public.profiles FOR SELECT
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Users can update own profile"
    ON public.profiles FOR UPDATE
    TO authenticated
    USING (id = auth.uid());

-- User roles: only admins can manage
CREATE POLICY "Admins can manage roles"
    ON public.user_roles FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

-- Quarters: readable by all authenticated users
CREATE POLICY "Authenticated users can view quarters"
    ON public.quarters FOR SELECT
    TO authenticated
    USING (true);

CREATE POLICY "Admins can manage quarters"
    ON public.quarters FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

-- Customers: admins can manage, customers can view own
CREATE POLICY "Admins can manage customers"
    ON public.customers FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Customers can view own record"
    ON public.customers FOR SELECT
    TO authenticated
    USING (user_id = auth.uid());

-- Customer quarter profiles: admins manage, customers view own
CREATE POLICY "Admins can manage customer quarter profiles"
    ON public.customer_quarter_profiles FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Customers can view own quarter profiles"
    ON public.customer_quarter_profiles FOR SELECT
    TO authenticated
    USING (
        customer_id IN (
            SELECT id FROM public.customers WHERE user_id = auth.uid()
        )
    );

-- Questions: readable by all authenticated users
CREATE POLICY "Authenticated users can view questions"
    ON public.questions FOR SELECT
    TO authenticated
    USING (true);

CREATE POLICY "Admins can manage questions"
    ON public.questions FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

-- Feedback assignments: admins manage all, customers view/update own
CREATE POLICY "Admins can manage feedback assignments"
    ON public.feedback_assignments FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Customers can view own assignments"
    ON public.feedback_assignments FOR SELECT
    TO authenticated
    USING (
        customer_id IN (
            SELECT id FROM public.customers WHERE user_id = auth.uid()
        )
    );

CREATE POLICY "Customers can update own pending assignments"
    ON public.feedback_assignments FOR UPDATE
    TO authenticated
    USING (
        customer_id IN (
            SELECT id FROM public.customers WHERE user_id = auth.uid()
        )
        AND status = 'pending'
    );

-- Feedback responses: admins view all, customers manage own
CREATE POLICY "Admins can view all feedback responses"
    ON public.feedback_responses FOR SELECT
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Customers can manage own responses"
    ON public.feedback_responses FOR ALL
    TO authenticated
    USING (
        assignment_id IN (
            SELECT fa.id FROM public.feedback_assignments fa
            JOIN public.customers c ON fa.customer_id = c.id
            WHERE c.user_id = auth.uid()
        )
    );

-- Feedback comments: same pattern as responses
CREATE POLICY "Admins can view all comments"
    ON public.feedback_comments FOR SELECT
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Customers can manage own comments"
    ON public.feedback_comments FOR ALL
    TO authenticated
    USING (
        assignment_id IN (
            SELECT fa.id FROM public.feedback_assignments fa
            JOIN public.customers c ON fa.customer_id = c.id
            WHERE c.user_id = auth.uid()
        )
    );

-- Quarter reports & dashboard metrics: readable by all authenticated
CREATE POLICY "Authenticated users can view quarter reports"
    ON public.quarter_reports FOR SELECT
    TO authenticated
    USING (true);

CREATE POLICY "Admins can manage quarter reports"
    ON public.quarter_reports FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

CREATE POLICY "Authenticated users can view dashboard metrics"
    ON public.dashboard_metrics FOR SELECT
    TO authenticated
    USING (true);

CREATE POLICY "Admins can manage dashboard metrics"
    ON public.dashboard_metrics FOR ALL
    TO authenticated
    USING (public.has_role(auth.uid(), 'admin'));

-- Function to calculate KPI percentage from average (scale 1-4, target mapped to 0-100%)
CREATE OR REPLACE FUNCTION public.avg_to_pct(avg_score NUMERIC)
RETURNS NUMERIC
LANGUAGE sql
IMMUTABLE
AS $$
    SELECT ROUND((avg_score / 4.0) * 100, 1)
$$;

-- Function to determine quarter outcome based on overall percentage
CREATE OR REPLACE FUNCTION public.determine_outcome(overall_pct NUMERIC)
RETURNS quarter_outcome
LANGUAGE sql
IMMUTABLE
AS $$
    SELECT CASE
        WHEN overall_pct >= 85.0 THEN 'incentive'::quarter_outcome
        WHEN overall_pct >= 80.0 THEN 'on_target'::quarter_outcome
        WHEN overall_pct >= 75.0 THEN 'below_target'::quarter_outcome
        ELSE 'penalty'::quarter_outcome
    END
$$;

-- ============================================================
-- TRIGGERS
-- ============================================================

-- Auto-update updated_at on profiles
CREATE OR REPLACE FUNCTION public.handle_updated_at()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    NEW.updated_at = now();
    RETURN NEW;
END;
$$;

CREATE TRIGGER on_profiles_updated
    BEFORE UPDATE ON public.profiles
    FOR EACH ROW
    EXECUTE FUNCTION public.handle_updated_at();

CREATE TRIGGER on_quarter_reports_updated
    BEFORE UPDATE ON public.quarter_reports
    FOR EACH ROW
    EXECUTE FUNCTION public.handle_updated_at();

-- Auto-create profile on new auth user signup
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
BEGIN
    INSERT INTO public.profiles (id, name, email)
    VALUES (
        NEW.id,
        COALESCE(NEW.raw_user_meta_data->>'name', split_part(NEW.email, '@', 1)),
        NEW.email
    );
    RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW
    EXECUTE FUNCTION public.handle_new_user();
