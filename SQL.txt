CREATE TYPE public.expat_type       AS ENUM ('new', 'existing');
CREATE TYPE public.feedback_status  AS ENUM ('pending', 'submitted');
CREATE TYPE public.quarter_outcome  AS ENUM ('on_target', 'below_target', 'penalty', 'incentive');
CREATE TYPE public.question_section AS ENUM ('service_initiation', 'service_delivery', 'driver_quality', 'overall');


-- ============================================================
-- TABLES
-- ============================================================

CREATE TABLE public.profiles (
    id         UUID         PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    name       VARCHAR(255) NOT NULL,
    email      VARCHAR(255) NOT NULL UNIQUE,
    created_at TIMESTAMPTZ  NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ  NOT NULL DEFAULT now()
);
GRANT ALL ON public.profiles TO authenticated;

-- ----

CREATE TABLE public.user_roles (
    id      UUID     PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID     NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    role    app_role NOT NULL,
    UNIQUE (user_id, role)
);
GRANT ALL ON public.user_roles TO authenticated;

-- ----

CREATE TABLE public.quarters (
    id             VARCHAR(20) PRIMARY KEY,
    label          VARCHAR(20) NOT NULL,
    year           INTEGER     NOT NULL,
    quarter_number INTEGER     NOT NULL CHECK (quarter_number BETWEEN 1 AND 4),
    is_active      BOOLEAN     NOT NULL DEFAULT false,
    created_at     TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE UNIQUE INDEX idx_quarters_single_active ON public.quarters (is_active) WHERE is_active = true;
GRANT ALL ON public.quarters TO authenticated;

-- ----

CREATE TABLE public.customers (
    id            SERIAL       PRIMARY KEY,
    name          VARCHAR(255) NOT NULL,
    email         VARCHAR(255) NOT NULL UNIQUE,
    phone         VARCHAR(50),
    employee_id   VARCHAR(20)  NOT NULL UNIQUE,
    is_active     BOOLEAN      NOT NULL DEFAULT true,
    created_at    DATE         NOT NULL DEFAULT CURRENT_DATE,
    allocated_car VARCHAR(100),
    start_date    DATE,
    end_date      DATE,
    user_id       UUID         REFERENCES auth.users(id) ON DELETE SET NULL
);
CREATE INDEX idx_customers_email   ON public.customers (email);
CREATE INDEX idx_customers_user_id ON public.customers (user_id);
GRANT ALL ON public.customers        TO authenticated;
GRANT ALL ON public.customers_id_seq TO authenticated;

-- ----

CREATE TABLE public.customer_quarter_profiles (
    id          SERIAL      PRIMARY KEY,
    customer_id INTEGER     NOT NULL REFERENCES public.customers(id)  ON DELETE CASCADE,
    quarter_id  VARCHAR(20) NOT NULL REFERENCES public.quarters(id)   ON DELETE CASCADE,
    expat_type  expat_type  NOT NULL DEFAULT 'existing',
    UNIQUE (customer_id, quarter_id)
);
CREATE INDEX idx_cqp_customer ON public.customer_quarter_profiles (customer_id);
CREATE INDEX idx_cqp_quarter  ON public.customer_quarter_profiles (quarter_id);
GRANT ALL ON public.customer_quarter_profiles        TO authenticated;
GRANT ALL ON public.customer_quarter_profiles_id_seq TO authenticated;

-- ----

CREATE TABLE public.questions (
    id                SERIAL           PRIMARY KEY,
    question_number   INTEGER          NOT NULL UNIQUE,
    text              TEXT             NOT NULL,
    section           question_section NOT NULL,
    is_new_expat_only BOOLEAN          NOT NULL DEFAULT false,

GRANT ALL ON public.questions        TO authenticated;
GRANT ALL ON public.questions_id_seq TO authenticated;

-- ----

CREATE TABLE public.feedback_assignments (
    id           SERIAL          PRIMARY KEY,
    quarter_id   VARCHAR(20)     NOT NULL REFERENCES public.quarters(id)  ON DELETE CASCADE,
    customer_id  INTEGER         NOT NULL REFERENCES public.customers(id) ON DELETE CASCADE,
    status       feedback_status NOT NULL DEFAULT 'pending',
    submitted_at TIMESTAMPTZ,
    created_at   TIMESTAMPTZ     NOT NULL DEFAULT now(),
    UNIQUE (quarter_id, customer_id)

CREATE INDEX idx_fa_customer ON public.feedback_assignments (customer_id);

GRANT ALL ON public.feedback_assignments        TO authenticated;
GRANT ALL ON public.feedback_assignments_id_seq TO authenticated;

-- ----

CREATE TABLE public.feedback_responses (

CREATE INDEX idx_fr_assignment ON public.feedback_responses (assignment_id);
GRANT ALL ON public.feedback_responses        TO authenticated;
GRANT ALL ON public.feedback_responses_id_seq TO authenticated;

-- ----

CREATE TABLE public.feedback_comments (

-- ============================================================
-- HELPER FUNCTIONS & TRIGGERS
-- ============================================================

-- Role check helper (used by the app to verify admin vs customer)
CREATE OR REPLACE FUNCTION public.has_role(_user_id UUID, _role app_role)
RETURNS BOOLEAN LANGUAGE sql STABLE SECURITY DEFINER SET search_path = public AS $$
    SELECT EXISTS (
        SELECT 1 FROM public.user_roles WHERE user_id = _user_id AND role = _role
    )
$$;

-- Auto-update updated_at
CREATE OR REPLACE FUNCTION public.handle_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$;

CREATE TRIGGER on_profiles_updated
    BEFORE UPDATE ON public.profiles
    FOR EACH ROW EXECUTE FUNCTION public.handle_updated_at();

CREATE TRIGGER on_quarter_reports_updated
    BEFORE UPDATE ON public.quarter_reports
    FOR EACH ROW EXECUTE FUNCTION public.handle_updated_at();

-- Auto-create profile row on auth signup
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$
BEGIN
    INSERT INTO public.profiles (id, name, email)
    VALUES (
        NEW.id,
        COALESCE(NEW.raw_user_meta_data->>'name', split_part(NEW.email, '@', 1)),
        NEW.email
    );
    RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
